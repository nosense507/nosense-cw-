#include <stdio.h>
char* strcpy(char *strDest, const char* strSrc)
{
    char *p=NULL;
    if(strDest == NULL || strSrc == NULL)
    {
        return NULL;
    }
    p = strDest;
    while((*strDest++ = *strSrc ++) != '\0');
    return p;
}
int getStrLen(const char* str)
{
    int len = 0;
    while( *str ++ != '\0')
    {
        len ++;
    }
    return len;
}
